@XmlAccessorType(XmlAccessType.FIELD)
public abstract class AbstractDAO<K,V> implements CRUD<K,V>{

  private static final long serialVersionUID = 1L;
  public HashMap<K,V> ddbb;

  public Boolean add(V v,K k) {
    boolean exists = false;
    if (v != null && k != null && !ddbb.containsKey(k)) {
      exists = true;
      ddbb.put(k, v);
    }
    return exists;
  }

  public V search(K k) {
    return ddbb.get(k);
  }

  public V delete(K k) {
    return ddbb.remove(k);
  }

  public Boolean update(V v, K k) {
    Boolean res = false;
    if(v != null && k != null && ddbb.containsKey(k)) {
      ddbb.put(k, v);
      res = true;
    }

    return res;
  }

  public abstract void save();

}

Abstract classes have allowed us to reduce to minimum repeated code, such is the case of the Data Access Objects who inherit from AbstractDAO. But most importantly, we have been able to interconnect all the controllers between each other, along with the Data Access Objects and all view objects thanks to the AbstractController class. The views have also access to the DAOs in order to make support consults.
This design is the most important single feature of the project and the results are the fact that we are able to use all functionalities of the program anywhere, without compromising encapsulation and the proper separation of the code. Thus, we can code efficiently, add new features light-fast and adapt to new requisites.
It should be mentioned that this last benefit is only possible if a proper system of interfaces have been set up in the design phase of the project.

public class Adapters {

  public static class LocalDateTimeBind extends XmlAdapter<String, LocalDateTime> {

    @Override
    public LocalDateTime unmarshal(String val) {
      return LocalDateTime.parse(val, DateTimeFormatter.ofPattern("yyyy-MM-dd HH:mm:ss:SS"));
    }

    @Override
    public String marshal(LocalDateTime localDateTime) {
      return localDateTime.format(DateTimeFormatter.ofPattern("yyyy-MM-dd HH:mm:ss:SS"));
    }
  }
}

      JAXBContext c = JAXBContext.newInstance(ItemDAO.class);
      Marshaller m = c.createMarshaller();
      m.setProperty(Marshaller.JAXB_FORMATTED_OUTPUT, true);

  }

public LocalDateTime readDateBucle(String mensaje){

public String readString(String mensaje,String regex)throws IllegalArgumentException {
  System.out.println(mensaje);
  String retCad = sc.nextLine();

  if(!retCad.matches(regex)) {
    throw new IllegalArgumentException("Formato invalido");
  }
  return retCad;
}

public String readStringBucle(String mensaje,String regex) {
  String retCad = null;
  Boolean invalido = true;
  do {
    invalido = true;
    try {
      retCad = this.readString(mensaje, regex);
      invalido=false;
    } catch (Exception e) {
      System.out.println(e.getMessage());
    }
  }while(invalido);
  return retCad;
}
